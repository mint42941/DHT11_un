    // ลิงก์โมเดลเดิมของคุณ
    const URL = "https://teachablemachine.withgoogle.com/models/CNTSCzzKD/";

    let model, webcam, labelContainer, maxPredictions;
    let predictTimer = null;

    // ฟังก์ชันโหลดโมเดลและตั้งค่ากล้องเว็บแคม
    async function init() {
        // อัปเดตปุ่มระหว่างโหลด
        const startBtn = document.querySelector('.btn-start');
        startBtn.innerText = "Loading...";
        startBtn.disabled = true;

        const modelURL = URL + "model.json";[cite: 2]
        const metadataURL = URL + "metadata.json";[cite: 2]

        model = await tmImage.load(modelURL, metadataURL);[cite: 2]
        maxPredictions = model.getTotalClasses();[cite: 2]

        const flip = true;[cite: 2]
        webcam = new tmImage.Webcam(200, 200, flip);[cite: 2]
        await webcam.setup();[cite: 2]
        await webcam.play();[cite: 2]
        window.requestAnimationFrame(loop);[cite: 2]

        // ซ่อนปุ่มเมื่อกล้องพร้อมทำงาน
        startBtn.style.display = "none";

        document.getElementById("webcam-container").appendChild(webcam.canvas);[cite: 2]
        labelContainer = document.getElementById("label-container");[cite: 2]
        labelContainer.innerHTML = ""; // ล้างค่าเก่าออกก่อน
        
        // สร้าง UI แถบความแม่นยำรอไว้
        for (let i = 0; i < maxPredictions; i++) {[cite: 2]
            const item = document.createElement("div");
            item.className = "prediction-item";
            item.innerHTML = `
                <div class="prediction-info">
                    <span class="class-name">-</span>
                    <span class="class-percent">0%</span>
                </div>
                <div class="progress-bar">
                    <div class="progress-fill"></div>
                </div>
            `;
            labelContainer.appendChild(item);
        }

        // สั่งให้ทำนายผลครั้งแรกทันที
        predict();

        // ตั้งเวลาให้เรียกใช้ฟังก์ชันทำนายผล (predict) ทุกๆ 3 วินาที (3000 มิลลิวินาที)
        if(predictTimer) clearInterval(predictTimer);
        predictTimer = setInterval(predict, 3000);
    }

    async function loop() {
        webcam.update(); // อัปเดตเฟรมภาพวิดีโอให้ลื่นไหลเป็นปกติ
        window.requestAnimationFrame(loop);[cite: 2]
    }

    // ฟังก์ชันทำนายผล (จะถูกเรียกใช้ทุกๆ 3 วินาทีตามที่กำหนดไว้ข้างบน)
    async function predict() {
        if (!model || !webcam) return;

        const prediction = await model.predict(webcam.canvas);[cite: 2]
        const items = labelContainer.getElementsByClassName("prediction-item");

        for (let i = 0; i < maxPredictions; i++) {[cite: 2]
            const className = prediction[i].className;[cite: 2]
            const probability = prediction[i].probability;[cite: 2]
            const percentText = (probability * 100).toFixed(0) + "%";

            // อัปเดตข้อความและแถบความมั่นใจ (Progress Bar)
            items[i].querySelector(".class-name").innerText = className;
            items[i].querySelector(".class-percent").innerText = percentText;
            items[i].querySelector(".progress-fill").style.width = percentText;
        }
    }
